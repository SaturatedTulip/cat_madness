function getRoundName(status) { const roundMap = { "1": "First Four", "2": "1st Round", "3": "2nd Round", "4": "Sweet 16", "5": "Elite 8", "6": "Final Four", "7": "Championship" }; if (roundMap[status.type.id]) { return roundMap[status.type.id]; } // Fallback to checking event notes headline const headline = status.notes ? status.notes.headline : ""; if (/Final Four/i.test(headline)) return "Final Four"; if (/Championship/i.test(headline)) return "Championship"; return "Tournament"; }
